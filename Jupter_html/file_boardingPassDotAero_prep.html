popular_cityes = ["MOSCOW", "SAINT PETERSBURG"]
pass_flights_to_city = None
pass_flights_df = pd.DataFrame(columns=["PASSENGER", "COUNT", "CITY"])
pass_flights_df
for city in popular_cityes:
    pass_flights_to_city = df[(df["FLIGHT_TO"] == city)]["PASS_NAME"].value_counts().to_frame()
    pass_flights_to_city = pass_flights_to_city.reset_index().rename(columns={"index": "PASSENGER", 
                                                       "PASS_NAME": "COUNT"})
    pass_flights_to_city["CITY"] = city
    pass_flights_df = pass_flights_df.append(pass_flights_to_city, ignore_index=True)
